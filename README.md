## POS (Windows)

~~~console
             
+------+     +------+     +------+
| COM2 | --- | ESPC | --- | COM1 |
+------+     +------+     +------+ 

POS print-out (com2 -> com1)
+------+     +------+     +------+
| COM2 | --> | ESPC | --> | COM1 |
+------+     +------+     +------+ 

PRINTER audit (com1 -> com2)
+------+     +------+     +------+
| COM2 | <-- | ESPC | <-- | COM1 |
+------+     +------+     +------+ 


~~~

~~~go

func main() {
	var wait sync.WaitGroup
	wait.Add(2)

	in := Open(IN)
	out := Open(OUT)

	// bi-direction communication.
	go Run(in, out)
	go Run(out, in)
	wait.Wait()
}

~~~
